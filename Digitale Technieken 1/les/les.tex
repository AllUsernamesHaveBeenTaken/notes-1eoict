\begin{verbatim}
A B X MINTERM
0 0 0 \overline{A}.\overline{B}
0 1 1 \overline{A}.B
1 0 1 A.\overline{B}
1 1 0 A.B
\end{verbatim}

$$X = A.\overline{B} + \overline{A}.B = A  \oplus  B$$

\end{verbatim}
